Theorem interp_deterministic : forall e v1 v2,
  interp e = v1 ->
  interp e = v2 ->
  v1 = v2.
Proof.
  intros. congruence.
Qed.

Theorem modulate_linear : forall m v1 v2,
  modulate m (zvector_plus v1 v2) =
  zvector_plus (modulate m v1) (modulate m v2).
Proof.
  intros. reflexivity.
Qed.
